\subsection{Moving Data Around}
1.size()命令，返回矩阵的尺寸
\begin{minted}{octave}
>> A=[1,2;3,4;5,6]
A =

   1   2
   3   4
   5   6

>> size(A)
#size()命令返回一个1*2的矩阵，返回矩阵的尺寸
ans =

   3   2

>> sz=size(A)
#这个矩阵用sz来存放，所以sz就是一个1*2的矩阵
sz =

   3   2

>> size(sz)
#计算矩阵的维度
ans =

   1   2

>> size(A,1)
#返回A矩阵的第一个元素3，行数
ans =  3
>> size(A,2)
#返回A矩阵的第2个元素2，列数
ans =  2
\end{minted}
2.length命令
\begin{minted}{octave}
>> v=[1 2 3 4]
#向量v

#一般只是给向量用length命令
ans =  5

\end{minted}
